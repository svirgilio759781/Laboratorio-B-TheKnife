CREATE TABLE Utente (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  -- Genera automaticamente ID univoci e sequenziali (Standard SQL)
    -- Quando definisci una colonna come IDENTITY, PostgreSQL crea automaticamente un oggetto nascosto chiamato Sequenza. Questa è essenzialmente un contatore che tiene traccia dell'ultimo numero utilizzato e sa che il prossimo dovrà essere +1.
    -- by default cosi in caso si può inserire il numero che si vuole
    -- si può cambiare manualmente l'id(se non è già esistente)
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    nome VARCHAR(30),
    cognome VARCHAR(30),
    dataNascita DATE,
    indirizzoDomicilio VARCHAR(255),
    cittaDomicilio VARCHAR(50),
    telefono VARCHAR(20),
    ruolo VARCHAR(30) CHECK (ruolo IN ('Admin', 'Cliente', 'Gestore')) NOT NULL -- Forza tipologia a essere Admin, Cliente o Gestore
);
CREATE TABLE Ristorante (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome TEXT NOT NULL,
    indirizzo TEXT NOT NULL,
    locazione VARCHAR(50) NOT NULL,
    prezzo VARCHAR(5) NOT NULL,
    tipologiaCucina VARCHAR(100) NOT NULL,
    latitudine DOUBLE PRECISION NOT NULL,
    longitudine DOUBLE PRECISION NOT NULL,
    telefono VARCHAR(20),
    url TEXT,
    websiteUrl TEXT,
    premi TEXT, -- TEXT non ha limite di caratteri
    greenStar INT check (greenStar >= 0 and greenStar <= 5),
    servizi TEXT,
    descrizione TEXT,
    consegnaDomicilio BOOLEAN DEFAULT FALSE,
    prenotazioneOnline BOOLEAN DEFAULT FALSE,
    idGestore INT SET DEFAULT 0,
    CONSTRAINT fk_gestore FOREIGN KEY (idGestore) REFERENCES Utente(id) -- uso constraint per semplificare in caso di modifica
        ON UPDATE CASCADE 
        ON DELETE SET NULL -- Se non si conosce più il gestore la riga rimane, ma con idGestore null
);
CREATE TABLE Preferito (
    idCliente INT,
    idRistorante INT,
    PRIMARY KEY (idCliente, idRistorante),
    CONSTRAINT fk_cliente_preferito FOREIGN KEY (idCliente) REFERENCES Utente(id)
        ON UPDATE CASCADE 
        ON DELETE CASCADE,
    CONSTRAINT fk_ristorante_preferito FOREIGN KEY (idRistorante) REFERENCES Ristorante(id)
        ON UPDATE CASCADE 
        ON DELETE CASCADE
);
CREATE TABLE Recensione (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    testo TEXT NOT NULL,
    stelle INT CHECK (stelle >= 1 AND stelle <= 5) NOT NULL,
    idCliente INT NOT NULL,
    idRistorante INT NOT NULL,
    CONSTRAINT fk_cliente_recensione FOREIGN KEY (idCliente) REFERENCES Utente(id)
        ON UPDATE CASCADE 
        ON DELETE CASCADE,
    CONSTRAINT fk_ristorante_recensione FOREIGN KEY (idRistorante) REFERENCES Ristorante(id)
        ON UPDATE CASCADE 
        ON DELETE CASCADE
);
CREATE TABLE RispostaRecensione (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, 
    testo TEXT NOT NULL,
    idRecensione INT UNIQUE NOT NULL, -- Solo una risposta per recensione
    CONSTRAINT fk_recensione_risposta FOREIGN KEY (idRecensione) REFERENCES Recensione(id)
        ON UPDATE CASCADE 
        ON DELETE CASCADE
);
CREATE TABLE Prenotazione (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    data DATE NOT NULL,
    ora TIME NOT NULL,
    numeroPersone INT CHECK (numero_persone > 0) NOT NULL,
    stato VARCHAR(50) DEFAULT 'In attesa',               -- Stato (In attesa, Confermata, Cancellata)
    idUtente INT NOT NULL,
    idRistorante INT NOT NULL,
    CONSTRAINT fk_utente_prenotazione FOREIGN KEY (idUtente) REFERENCES Utente(id)
        ON UPDATE CASCADE 
        ON DELETE CASCADE,
    CONSTRAINT fk_ristorante_prenotazione FOREIGN KEY (idRistorante) REFERENCES Ristorante(id)
        ON UPDATE CASCADE 
        ON DELETE CASCADE
);
