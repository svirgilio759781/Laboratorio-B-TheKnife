-- Tabella Utente
-- Serial: Crea un oggetto Sequence (un contatore separato),Imposta il valore predefinito della colonna in modo che prelevi il numero successivo da quella sequenza.
CREATE TABLE utente (
    	id SERIAL PRIMARY KEY,
    	username VARCHAR(50) NOT NULL UNIQUE,
    	password VARCHAR(255) NOT NULL,
	nome VARCHAR(30) NOT NULL,
	cognome VARCHAR(30) NOT NULL,
	dataNascita DATE NOT NULL,
	indirizzoDomicilio VARCHAR(255) NOT NULL,
	cittaDomicilio VARCHAR(50) NOT NULL,
    	tipologia VARCHAR(20) CHECK (tipologia IN ('Cliente', 'Gestore')) NOT NULL
);
